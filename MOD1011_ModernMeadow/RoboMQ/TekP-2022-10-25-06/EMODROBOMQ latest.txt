BEGIN

    DECLARE  @FormatCode        VARCHAR(10)

    -- Set FormatCode

         @StartPerControl = StartPerControl

    DELETE FROM dbo.U_EMODROBOMQ_EEList
    WHERE xCoID <> dbo.dsi_BDM_fn_GetCurrentCOID(xEEID)
    AND xEEID IN (SELECT xEEID FROM dbo.U_EMODROBOMQ_EEList GROUP BY xEEID HAVING COUNT(1) > 1);

    --==========================================
    -- Build Driver Tables
    --==========================================
    ---------------------------------
    -- DETAIL RECORD - U_EMODROBOMQ_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EMODROBOMQ_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EMODROBOMQ_drvTbl;
   SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvSort = EecEmpNo
        -- standard fields above and additional driver fields below
        ,drvEmplStatus = EecEmplStatus
        ,drvEmplNo = EecEmpNo
        ,drvNameFirst = EepNameFirst
        ,drvPreferredName = EepNamePreferred
        ,drvNameLast = EepNameLast
        ,drvPreferredLast = EepUDField01
        ,drvJob = JbcDesc
        ,drvOrgLevel1 = Org1.OrgDesc
        ,drvSupervisorNo = dbo.dsi_fnlib_GetSupervisorField_v2('', dbo.vw_int_EmpComp.EecEEID, 'EmpNo') 
        ,drvOrgLevel2 = Org2.OrgDesc
        ,drvHireDate = CASE WHEN EecDateOfLastHire<>EecDateofOriginalHire THEN EecDateOfLastHire else EecDateofOriginalHire END
        ,drvJobStartDate = EecEmplStatusStartDate
        ,drvTerminationDate = CASE WHEN EecEmplStatus = 'T' THEN EecDateOfTermination END
    INTO dbo.U_EMODROBOMQ_drvTbl
    FROM dbo.U_EMODROBOMQ_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN dbo.JobCode WITH (NOLOCK)
        ON JbcJobCode = EecJobCode
    LEFT JOIN dbo.OrgLevel Org2 WITH (NOLOCK)
        ON Org2.OrgCode = EecOrgLvl2
        and Org2.OrgLvl = '2'
    LEFT JOIN dbo.OrgLevel Org1 WITH (NOLOCK)
        ON Org1.OrgCode = EecOrgLvl1
        and Org1.OrgLvl = '1'
    ;


    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable(@FormatCode,'UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable(@FormatCode,'Testing') = 'Y' THEN 'Test_Filename_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                                  ELSE 'Filename_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                             END
        WHERE FormatCode = @FormatCode;
    END

END;
