    EXEC dbo.dsi_BDM_sp_PopulateDeductionsTable @FormatCode;
	EXEC dbo.dsi_bdm_sp_CalculateBenefitAmounts @FormatCode;

    -----------------------------
    -- Working Table - PDedHist
    -----------------------------
    IF OBJECT_ID('U_ESHEROCHSA_PDedHist','U') IS NOT NULL
        DROP TABLE dbo.U_ESHEROCHSA_PDedHist;
    SELECT DISTINCT
         PdhEEID
        -- Current Payroll Amounts
        ,PdhEECurAmt    = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhERCurAmt    = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl THEN PdhERCurAmt ELSE 0.00 END)
        -- YTD Payroll Amounts
        ,PdhEECurAmtYTD = SUM(PdhEECurAmt)
        ,PdhERCurAmtYTD = SUM(PdhERCurAmt)
        -- Categorize Payroll Amounts
        ,PdhHSA     = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl AND PdhDedCode IN ('HSAI','HSAF','HSACI','HSACF') THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhHASR    = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl AND PdhDedCode IN ('HASER') THEN PdhEECurAmt ELSE 0.00 END)
    INTO dbo.U_ESHEROCHSA_PDedHist
    FROM dbo.PDedHist WITH (NOLOCK)
    JOIN dbo.U_ESHEROCHSA_DedList WITH (NOLOCK)
        ON DedCode = PdhDedCode
    WHERE LEFT(PdhPerControl,4) = LEFT(@EndPerControl,4)
    AND PdhPerControl <= @EndPerControl
    --AND PdhPerControl BETWEEN @StartPerControl AND @EndPerControl -- Filter for Current Payroll Dates. If you need YTD Totals, then remove or comment out this line.
    GROUP BY PdhEEID
    HAVING (SUM(PdhEECurAmt) <> 0.00
        OR SUM(PdhERCurAmt) <> 0.00
    );


    --==========================================
    -- Build Driver Tables
    --==========================================
    ---------------------------------
    -- DETAIL RECORD - U_ESHEROCHSA_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_ESHEROCHSA_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_ESHEROCHSA_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvInitialSort = ''
        -- standard fields above and additional driver fields below
        ,drvNameFirst = EepNameFirst
        ,drvNameLast = EepNameLast
        ,drvEmpNo = EecEmpNo
        ,drvAddress = EepAddressLine1 
        ,drvCity = EepAddressCity
        ,drvState = EepAddressState
        ,drvZip = EepAddressZipCode
        ,drvEmail = EepAddressEMail
        ,drvDateofBirth = EepDateOfBirth
        ,drvPhoneNumber = CASE WHEN EepPhoneHomeNumber is null then dbo.dsi_fnlib_GetPhoneNumber(EepEEID, 'CEL') else EepPhoneHomeNumber END
        ,drvEEPerPay = FORMAT(PdhHSA ,'##0.00')
        ,drvERPerPay = FORMAT(PdhHASR ,'##0.00')
        ,drvAnnualElection = FORMAT(BdmEEGoalAmt,'##0.00')
        ,drvCurrentEffectiveDate = CASE WHEN EecEmplStatus <> 'T' THEN '01/01/2024' END
        ,drvPayrollDate = ''
        ,drvTerminationDate = CASE WHEN EecEmplStatus = 'T' AND EecDateOfTermination > '2023-12-17' THEN EecDateOfTermination END
        ,drvLastContribution = ''
        ,drvSSN = EepSSN
        ,drvPlanYearStart = '01/01/2024'
        ,drvPlanYearEnd = '12/31/2024'
    INTO dbo.U_ESHEROCHSA_drvTbl
    FROM dbo.U_ESHEROCHSA_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
        AND EecEEID not in ('ENT5JS000RW0','EL4MWL000RW0','ELJFVK000RW0','EKU2PT000RW0')
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    LEFT JOIN dbo.U_ESHEROCHSA_PDedHist WITH (NOLOCK)
        ON PdhEEID = xEEID
    JOIN dbo.U_dsi_BDM_ESHEROCHSA WITH (NOLOCK)
        On BdmEEID = xEEID
        AND BdmCoID = xCoID
    --JOIN (Select EedEEID, EedCoID
 --               ,MAX(DedDedType) AS DedType_HSA                      
 --               ,MAX(EedBenStartDate) AS StartDate_HSA   
 --               ,MAX(EedBenStopDate) AS StopDate_HSA 
 --               FROM dbo.EmpDed WITH (NOLOCK)
    --            JOIN dbo.DedCode WITH (NOLOCK)
    --            On EedDedCode = DedDedCode
 --               Where DedDedType in ('HSA')
    --            AND EedBenStopDate> @StartDate or EedBenStopDate is null
 --               GROUP BY EedEEID, EedCoID) 
    --    HSATable
    --    ON EedEEID = xEEID
    --    AND EedCOID = xCOID
    ;

    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable(@FormatCode,'UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable(@FormatCode,'Testing') = 'Y' THEN 'SHE_TEST_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.csv'
                                  WHEN @ExportCode LIKE 'OE%' THEN 'OE_SHE_TEST_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.csv'
                                  ELSE 'SHE_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.csv'
                             END
        WHERE FormatCode = @FormatCode;
    END

END;
