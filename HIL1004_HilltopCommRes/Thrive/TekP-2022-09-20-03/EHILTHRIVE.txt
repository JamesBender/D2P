CREATE PROCEDURE [dbo].[dsi_sp_BuildDriverTables_EHILTHRIVE]
    @SystemID char(12)
AS
SET NOCOUNT ON;
/**********************************************************************************
Client Name: Hilltop Community Resources, Inc.

Created By: Marie Thomerson
Business Analyst: Curlin Beck
Create Date: 01/04/2023
Service Request Number: TekP-2022-09-20-03

Purpose: Thrive Pass FSA Export

Revision History

Update By           Date           Request Num        Desc
XXXX                XX/XX/2023     SR-2023-000XXXXX   XXXXX

SELECT * FROM dbo.U_dsi_Configuration WHERE FormatCode = 'EHILTHRIVE';
SELECT * FROM dbo.U_dsi_SqlClauses WHERE FormatCode = 'EHILTHRIVE';
SELECT * FROM dbo.U_dsi_Parameters WHERE FormatCode = 'EHILTHRIVE';
SELECT ExpFormatCode, ExpExportCode, ExpStartPerControl, ExpEndPerControl,* FROM dbo.AscExp WHERE expFormatCode = 'EHILTHRIVE';
SELECT * FROM dbo.U_dsi_InterfaceActivityLog WHERE FormatCode = 'EHILTHRIVE' ORDER BY RunID DESC;


--Update Dates
UPDATE dbo.AscExp
    SET expLastStartPerControl = '202202281'
       ,expStartPerControl     = '202202281'
       ,expLastEndPerControl   = '202301049'
       ,expEndPerControl       = '202301049'
WHERE expFormatCode = 'EHILTHRIVE';

Execute Export
--------------
EXEC dbo.dsi_sp_TestSwitchbox_v2 'EHILTHRIVE', 'ONDEM_XOE';
EXEC dbo.dsi_sp_TestSwitchbox_v2 'EHILTHRIVE', 'TEST_XOE';
EXEC dbo.dsi_sp_TestSwitchbox_v2 'EHILTHRIVE', 'OEPASSIVE';
EXEC dbo.dsi_sp_TestSwitchbox_v2 'EHILTHRIVE', 'OEACTIVE';
EXEC dbo.dsi_sp_TestSwitchbox_v2 'EHILTHRIVE', 'SCH_EHILTH';

EXEC dbo.dsi_BDM_sp_ErrorCheck 'EHILTHRIVE';

EXEC dbo._dsi_usp_ExportRipOut_v7_4 @FormatCode = 'EHILTHRIVE', @AllObjects = 'Y', @IsWeb = 'Y'
**********************************************************************************/
BEGIN

    --==========================================
    -- Declare variables
    --==========================================
    DECLARE  @FormatCode        VARCHAR(10)
            ,@ExportCode        VARCHAR(10)
            ,@StartDate         DATETIME
            ,@EndDate           DATETIME
            ,@StartPerControl   VARCHAR(9)
            ,@EndPerControl     VARCHAR(9);

    -- Set FormatCode
    SELECT @FormatCode = 'EHILTHRIVE';

    -- Declare dates from Parameter file
    SELECT
         @StartPerControl = StartPerControl
        ,@EndPerControl   = EndPerControl
        ,@StartDate       = LEFT(StartPerControl,8)
        ,@EndDate         = DATEADD(S,-1,DATEADD(D,1,LEFT(EndPerControl,8)))
        ,@ExportCode      = ExportCode
    FROM dbo.U_dsi_Parameters WITH (NOLOCK)
    WHERE FormatCode = @FormatCode;

    --==========================================
    -- Clean EE List 
    -- Caution: Careful of cleaning EE List if including paycheck data
    --==========================================

    -- Cleans EE List of terms where EE active in another company (transfer), or active in more than one company
    DELETE FROM dbo.U_EHILTHRIVE_EEList
    WHERE xCoID <> dbo.dsi_BDM_fn_GetCurrentCOID(xEEID)
    AND xEEID IN (SELECT xEEID FROM dbo.U_EHILTHRIVE_EEList GROUP BY xEEID HAVING COUNT(1) > 1);

    --==========================================
    -- Audit Section
    --==========================================
    -- Get data from audit fields table. Add fields here if auditing
    IF OBJECT_ID('U_EHILTHRIVE_AuditFields','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_AuditFields;
    CREATE TABLE dbo.U_EHILTHRIVE_AuditFields (aTableName varchar(30),aFieldName varchar(30));

    INSERT INTO dbo.U_EHILTHRIVE_AuditFields VALUES ('Empcomp','eecEmplStatus');

    -- Create audit table based on fields defined above
    IF OBJECT_ID('U_EHILTHRIVE_Audit','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_Audit;
    SELECT 
        audEEID  = audKey1Value
        ,audKey2 = audKey2Value
        ,audKey3 = audKey3Value
        ,audTableName
        ,audFieldName
        ,audAction
        ,audDateTime
        ,audOldValue
        ,audNewValue
        ,audRowNo = ROW_NUMBER() OVER (PARTITION BY audKey1Value, audKey2Value, audKey3Value, audFieldName ORDER BY audDateTime DESC)
    INTO dbo.U_EHILTHRIVE_Audit
    FROM dbo.vw_AuditData WITH (NOLOCK) 
    JOIN dbo.U_EHILTHRIVE_AuditFields WITH (NOLOCK) 
        ON audTableName = aTableName
        AND audFieldName = aFieldName
    WHERE audDateTime BETWEEN @StartDate AND @EndDate
    AND audAction <> 'DELETE';

    -- Create Index
    CREATE CLUSTERED INDEX CDX_U_EHILTHRIVE_Audit ON dbo.U_EHILTHRIVE_Audit (audEEID,audKey2);

    --==========================================
    -- Create Deduction List
    --==========================================
    DECLARE @DedList VARCHAR(MAX)
    SET @DedList = '106,107';

    IF OBJECT_ID('U_EHILTHRIVE_DedList','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_DedList;
    SELECT DISTINCT
         DedCode = DedDedCode
        ,DedType = DedDedType
    INTO dbo.U_EHILTHRIVE_DedList
    FROM dbo.fn_ListToTable(@DedList)
    JOIN dbo.DedCode WITH (NOLOCK)
        ON DedDedCode = Item;

    --==========================================
    -- BDM Section
    --==========================================
    DELETE FROM dbo.U_dsi_BDM_Configuration WHERE FormatCode = @FormatCode;

    -- Required parameters
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'DedCodes','106, 107');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'StartDateTime',@StartDate);
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'EndDateTime',@EndDate);
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'TermSelectionOption','AuditDate');

    -- Non-Required parameters
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'BuildConsolidatedTable','Standard');
    --INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'CobraType','4');
    --INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'CobraDate','3');
    --INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsSpouse','SPS');
    --INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsChild','CHL');
    --INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsDomPartner','DP');

    -- Required OE parameters
    IF @ExportCode LIKE '%PASSIVE'
    BEGIN
        INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'OEType','PASSIVE');
    END;

    IF @ExportCode LIKE '%ACTIVE'
    BEGIN
        INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'OEType','ACTIVE');
    END;


    -- Run BDM Module
    EXEC dbo.dsi_BDM_sp_PopulateDeductionsTable @FormatCode;

    --==========================================
    -- Build Working Tables
    --==========================================

    -----------------------------
    -- Working Table - PDedHist
    -----------------------------
    IF OBJECT_ID('U_EHILTHRIVE_PDedHist','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_PDedHist;
    SELECT DISTINCT
         PdhEEID
        -- Current Payroll Amounts
        ,PdhEECurAmt    = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhERCurAmt    = SUM(CASE WHEN PdhPerControl BETWEEN @StartPerControl AND @EndPerControl THEN PdhERCurAmt ELSE 0.00 END)
        -- YTD Payroll Amounts
        ,PdhEECurAmtYTD = SUM(PdhEECurAmt)
        ,PdhERCurAmtYTD = SUM(PdhERCurAmt)
        -- Categorize Payroll Amounts
        ,PdhSource107   = SUM(CASE WHEN PdhDedCode IN ('107') THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhSource106   = SUM(CASE WHEN PdhDedCode IN ('106') THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhSource3     = SUM(CASE WHEN PdhDedCode IN ('106','107') THEN PdhEECurAmt ELSE 0.00 END)        
        ,PdhSource4     = SUM(CASE WHEN PdhDedCode IN ('401CU') THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhSource5     = SUM(CASE WHEN PdhDedCode IN ('ROTHC') THEN PdhEECurAmt ELSE 0.00 END)
        ,PdhSource6     = SUM(CASE WHEN PdhDedCode IN ('401KL1') THEN ISNULL(PdhEECurAmt, 0) ELSE 0.00 END)
        ,PdhSource7     = SUM(CASE WHEN PdhDedCode IN ('401KL2') THEN ISNULL(PdhEECurAmt, 0) ELSE 0.00 END)
        ,PdhSource8     = SUM(CASE WHEN PdhDedCode IN ('401KL3') THEN ISNULL(PdhEECurAmt, 0) ELSE 0.00 END)
        ,PdhSource9     = SUM(CASE WHEN PdhDedCode IN ('401KL4') THEN ISNULL(PdhEECurAmt, 0) ELSE 0.00 END)
        ,PdhSource10    = SUM(CASE WHEN PdhDedCode IN ('401KL5') THEN ISNULL(PdhEECurAmt, 0) ELSE 0.00 END)
    INTO dbo.U_EHILTHRIVE_PDedHist
    FROM dbo.PDedHist WITH (NOLOCK)
    --JOIN dbo.U_EHILTHRIVE_DedList WITH (NOLOCK)
    --    ON DedCode = PdhDedCode
    WHERE LEFT(PdhPerControl,4) = LEFT(@EndPerControl,4)
    AND PdhPerControl <= @EndPerControl
    AND PdhPerControl BETWEEN @StartPerControl AND @EndPerControl -- Filter for Current Payroll Dates. If you need YTD Totals, then remove or comment out this line.
    GROUP BY PdhEEID
    HAVING (SUM(PdhEECurAmt) <> 0.00
        OR SUM(PdhERCurAmt) <> 0.00
    );


    -----------------------------
    -- Working Table - PEarHist
    -----------------------------
    IF OBJECT_ID('U_EHILTHRIVE_PEarHist','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_PEarHist;
    SELECT DISTINCT
         PehEEID
        ,PrgPayDate             = MAX(PrgPayDate)
        -- Current Payroll Amount/Hours
        ,PehCurAmt              = SUM(CASE WHEN PehPerControl >= @StartPerControl THEN PehCurAmt ELSE 0.00 END)
        ,PehCurHrs              = SUM(CASE WHEN PehPerControl >= @StartPerControl THEN PehCurHrs ELSE 0.00 END)
        -- YTD Payroll Amount/Hours
        ,PehCurAmtYTD           = SUM(PehCurAmt)
        ,PehCurHrsYTD           = SUM(PehCurHrs)
        -- Current Include Deferred Comp Amount/Hours
        ,PehInclInDefComp       = SUM(CASE WHEN PehInclInDefComp = 'Y' AND PehPerControl >= @StartPerControl THEN PehCurAmt END)
        ,PehInclInDefCompHrs    = SUM(CASE WHEN PehInclInDefCompHrs = 'Y' AND PehPerControl >= @StartPerControl THEN PehCurHrs END)
        -- YTD Include Deferred Comp Amount/Hours
        ,PehInclInDefCompYTD    = SUM(CASE WHEN PehInclInDefComp = 'Y' THEN PehCurAmt END)
        ,PehInclInDefCompHrsYTD = SUM(CASE WHEN PehInclInDefCompHrs = 'Y' THEN PehCurHrs END)
    INTO dbo.U_EHILTHRIVE_PEarHist
    FROM dbo.vw_int_PayReg WITH (NOLOCK)
    JOIN dbo.vw_int_PEarHist WITH (NOLOCK)
        ON PehGenNumber = PrgGenNumber
    WHERE LEFT(PehPerControl,4) = LEFT(@EndPerControl,4)
    AND PehPerControl <= @EndPerControl
    GROUP BY PehEEID
    HAVING SUM(PehCurAmt) <> 0.00;
    --==========================================
    -- Build Driver Tables
    --==========================================
    ---------------------------------
    -- DETAIL RECORD - U_EHILTHRIVE_IB_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EHILTHRIVE_IB_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_IB_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvEmpNoSort = eepSSN
        -- standard fields above and additional driver fields below
        ,drvTPAID = CASE WHEN @ExportCode LIKE '%TEST%' THEN 'T01611' ELSE 'T02104' END   -- Updated TPA ID values CB 20230112
        ,drvSSN = eepSSN
        ,drvNameLast = EepNameLast
        ,drvNameFirst = EepNameFirst
        ,drvNameMiddle = LEFT(EepNameMiddle,1)
        ,drvHomePhone = EepPhoneHomeNumber
        ,drvCel = dbo.dsi_fnlib_GetPhoneNumber(EepEEID, 'CEL') 
        ,drvAddressLine1 = EepAddressLine1
        ,drvAddressLine2 = EepAddressLine2
        ,drvAddressCity = EepAddressCity
        ,drvAddressState = EepAddressState
        ,drvAddressZipCode = EepAddressZipCode
        ,drvEmployeeStatus = CASE WHEN EecEmplStatus = 'T' then '5' else '2' END
        ,drvGenderGender = CASE WHEN EepGender = 'M' THEN '1' 
                                WHEN EepGender = 'F' THEN '2'
                                ELSE '0' END
        ,drvMaritalStatus = CASE when EepMaritalStatus  = 'S' THEN '1'
                                when EepMaritalStatus  = 'M' THEN '2'
                                when EepMaritalStatus  = 'W' THEN '4'
                                when EepMaritalStatus  = 'D' THEN '5'
                                Else '0' END
        ,drvBirthDate = EepDateOfBirth
        ,drvEmployeeNumber = EepSSN
        ,drvDateofLastHire = EecDateOfLastHire
        ,drvTerminationDate = CASE WHEN EecEmplStatus = 'T' THEN EecDateOfTermination END
    INTO dbo.U_EHILTHRIVE_IB_drvTbl
    FROM dbo.U_EHILTHRIVE_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
                 And (eecemplstatus <> 'T' OR (eecemplstatus = 'T' and eectermreason <> 'TRO'
          and EXISTS (Select 1 from dbo.U_EHILTHRIVE_Audit where audEEID = xEEID and audfieldname = 'eecemplstatus' and audNewValue = 'T')))
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN dbo.U_dsi_BDM_EHILTHRIVE WITH (NOLOCK)
        ON BdmEEID = xEEID 
        AND BdmCoID = xCoID
        --AND BdmRecType = 'EMP'
    ;
    ---------------------------------
    -- DETAIL RECORD - U_EHILTHRIVE_IC_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EHILTHRIVE_IC_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_IC_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvEmpNoSort = eepSSN
        -- standard fields above and additional driver fields below
        ,drvTPAID = CASE WHEN @ExportCode LIKE '%TEST%' THEN 'T01611' ELSE 'T02104' END   -- Updated TPA ID values CB 20230112
        ,drvSSN = eepSSN
        ,drvPlanID = CASE    WHEN BdmDedCode = '107' THEN 'DCA' 
                            WHEN BdmDedCode = '106' THEN 'FSA' END
        ,drvAccountTypeCode = CASE    WHEN BdmDedCode = '107' THEN 'DCA' 
                                    WHEN BdmDedCode = '106' THEN 'FSA' END
        ,drvPlanStartDate =  CASE WHEN month(@StartDate) > 6 THEN CAST(year(@StartDate) as VARCHAR(4)) + '0701' WHEN month(@StartDate) < 7 THEN CAST(year(@StartDate)-1 as VARCHAR(4)) + '0701'  END
        ,drvPlanEndDate =  CASE WHEN month(@StartDate) > 5 THEN CAST(year(@StartDate)+1 as VARCHAR(4)) + '0630' WHEN month(@StartDate) < 6 THEN CAST(year(@StartDate) as VARCHAR(4)) + '0630'  END
        ,drvAccountStatus = CASE WHEN EecEmplStatus = 'T' THEN '5' ELSE '2' END
        ,drvOriginalPrefundedAmt = BdmEEGoalAmt
        ,drvEffectiveDate = BdmBenStartDate
        ,drvTerminationDate = CASE WHEN EecEmplStatus = 'T' THEN EecDateOfTermination END
        ,drvEmployerAnnElection = CASE WHEN BdmDedCode = '107' then (BdmEEGoalAmt)/2 else '0.00' END
    INTO dbo.U_EHILTHRIVE_IC_drvTbl
    FROM dbo.U_EHILTHRIVE_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
                         And (eecemplstatus <> 'T' OR (eecemplstatus = 'T' and eectermreason <> 'TRO'
          and EXISTS (Select 1 from dbo.U_EHILTHRIVE_Audit where audEEID = xEEID and audfieldname = 'eecemplstatus' and audNewValue = 'T')))
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN dbo.U_dsi_BDM_EHILTHRIVE WITH (NOLOCK)
        ON BdmEEID = xEEID 
        AND BdmCoID = xCoID
    ;
    ---------------------------------
    -- DETAIL RECORD - U_EHILTHRIVE_IH_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EHILTHRIVE_IH_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_IH_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvEmpNoSort = eepSSN
        -- standard fields above and additional driver fields below
        ,drvTPAID = CASE WHEN @ExportCode LIKE '%TEST%' THEN 'T01611' ELSE 'T02104' END   -- Updated TPA ID values CB 20230112 
        ,drvSSN = eepSSN
        ,drvPlanID = CASE    WHEN BdmDedCode = '107' THEN 'DCA' 
                            WHEN BdmDedCode = '106' THEN 'FSA' END
        ,drvAccountTypeCode = CASE    WHEN BdmDedCode = '107' THEN 'DCA' 
                                    WHEN BdmDedCode = '106' THEN 'FSA' END
        --,drvPlanStartDate =  CAST(year(GetDate()) as VARCHAR(4)) + '0701'              --Corrected the date format to YYYYMMDD CB 01/12/2023
        --,drvPlanEndDate =    CAST((year(GetDate())+1) as VARCHAR(4)) + '0630'          --Corrected the date format to YYYYMMDD CB 01/12/2023
        ,drvPlanStartDate =  CASE WHEN month(@StartDate) > 6 THEN CAST(year(@StartDate) as VARCHAR(4)) + '0701' WHEN month(@StartDate) < 7 THEN CAST(year(@StartDate)-1 as VARCHAR(4)) + '0701'  END
        ,drvPlanEndDate =  CASE WHEN month(@StartDate) > 5 THEN CAST(year(@StartDate)+1 as VARCHAR(4)) + '0630' WHEN month(@StartDate) < 6 THEN CAST(year(@StartDate) as VARCHAR(4)) + '0630'  END

        ,drvEmployeeDepositAmt = FORMAT(PdhSource3,'0.00')
        ,drvEmployerDepositAmt = CASE WHEN BdmDedCode = '107' THEN FORMAT(PdhSource107,'0.00') ELSE '0.00' END
        ,drvEffectiveDate = PrgPaydate
    INTO dbo.U_EHILTHRIVE_IH_drvTbl
    FROM dbo.U_EHILTHRIVE_EEList WITH (NOLOCK)
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
                                 And (eecemplstatus <> 'T' OR (eecemplstatus = 'T' and eectermreason <> 'TRO'
          and EXISTS (Select 1 from dbo.U_EHILTHRIVE_Audit where audEEID = xEEID and audfieldname = 'eecemplstatus' and audNewValue = 'T')))
    JOIN dbo.U_dsi_BDM_EHILTHRIVE WITH (NOLOCK)
        ON BdmEEID = xEEID 
        AND BdmCoID = xCoID
    LEFT JOIN dbo.U_EHILTHRIVE_PDedHist WITH (NOLOCK)
        ON PdhEEID = xEEID
    LEFT JOIN dbo.U_EHILTHRIVE_PEarHist WITH (NOLOCK)
        ON PehEEID = xEEID
    ;
    ---------------------------------
    -- HEADER RECORD
    ---------------------------------
    IF OBJECT_ID('U_EHILTHRIVE_HDR_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EHILTHRIVE_HDR_drvTbl;
    SELECT DISTINCT
         drvNumberofRows = count(drvEEID)+1
        ,drvEDIPassword = CASE WHEN @ExportCode LIKE '%TEST%' THEN 'T01611' ELSE 'T02104' END   -- Updated TPA ID values CB 20230112
    INTO dbo.U_EHILTHRIVE_HDR_drvTbl
    FROM dbo.U_EHILTHRIVE_IB_drvTbl WITH (NOLOCK)
    --JOIN dbo.U_dsi_BDM_EHILTHRIVE WITH (NOLOCK)
    --    ON BdmEEID = xEEID 
    --    AND BdmCoID = xCoID
    ;

    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable(@FormatCode,'UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable(@FormatCode,'Testing') = 'Y' THEN 'TEST_HILLTOP_PRETAX_IBICIH_' + REPLACE(CONVERT(VARCHAR(10),GETDATE(),101),'/','') + '_hhmmss.mbi'
                                  ELSE 'HILLTOP_PRETAX_IBICIH_' + REPLACE(CONVERT(VARCHAR(10),GETDATE(),101),'/','') + '_hhmmss.mbi'
                             END
        WHERE FormatCode = @FormatCode;
    END

END;
