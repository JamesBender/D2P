                                        WHEN efoPhoneType = 'CEL' AND efoPhoneNumber is not null then 'MOBILE' 
                                        ELSE '  ' 
                                        END
        ,drvAREACODE = CASE    WHEN EP.EepPhoneHomeNumber is not null THEN LEFT(EP.EepPhoneHomeNumber,3)
                                        WHEN efoPhoneType = 'CEL' AND efoPhoneNumber is not null then LEFT(efoPhoneNumber,3)
                                        ELSE '   ' 
                                        END
        ,drvTELEPHONENUMBER = CASE    WHEN EP.EepPhoneHomeNumber is not null THEN RIGHT(EP.EepPhoneHomeNumber,7)
                                        WHEN efoPhoneType = 'CEL' AND efoPhoneNumber is not null then RIGHT(efoPhoneNumber,7)
                                        ELSE '                                   ' 
                                        END
        ,drvSMSCAPABLEINDICATOR = 'N'
        ,drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvInitialSort = RTRIM(EP.EepSSN)
        ,drvSubSort = '16'
    INTO dbo.U_EFDSNELEXP_drvTbl_PH
    FROM dbo.U_EFDSNELEXP_Employees2 WITH (NOLOCK)
        LEFT JOIN dbo.U_dsi_BDM_EFDSNELEXP WITH (NOLOCK)
        ON BdmEEID = xEEID
        AND BdmCOID = xCOID
    --JOIN dbo.EmpComp EC WITH (NOLOCK)
    --    On EecEEID = xEEID
    --    AND EecCOID = xCOID
    JOIN dbo.EmpPers EP WITH (NOLOCK)
        On EepEEID = xEEID
    --JOIN dbo.EmpDed WITH (NOLOCK)
    --    ON EedEEID = bdmEEID 
    --    AND EedCoID = BdmCoID
    --    and EedDedCode = BdmDedCode
    LEFT JOIN dbo.EmpMPhon WITH (NOLOCK)
        ON efoEEID = xEEID
        ;

    INSERT INTO dbo.U_EFDSNELEXP_drvTbl_PH
    SELECT DISTINCT
         drvPLANNUMBER = '54099' 
        ,drvSOCIALSECURITYNUMBER 
        ,drvTELEPHONETYPECODE
        ,drvAREACODE
        ,drvTELEPHONENUMBER
        ,drvSMSCAPABLEINDICATOR
        ,drvEEID
        ,drvCoID
        ,drvInitialSort 
        ,drvSubSort 
    FROM  dbo.U_EFDSNELEXP_drvTbl_PH
    WHERE drveeid in (select bdmeeid from dbo.U_dsi_BDM_EFDSNELEXP where bdmdedcode in ('409AP', '409NF', '40F', '40SH') )
    


    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable('EFDSNELEXP','UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable('EFDSNELEXP','Testing') = 'Y' THEN 'TEST_CSP_EFDSNELEXP_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                                  ELSE 'CSP_EFDSNELEXP_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                             END
        WHERE FormatCode = 'EFDSNELEXP';
    END;

END;
