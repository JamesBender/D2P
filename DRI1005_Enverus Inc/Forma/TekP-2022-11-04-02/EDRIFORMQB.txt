BEGIN

    --==========================================

    --==========================================

    -- Set FormatCode
    SELECT @FormatCode = 'EDRIFORMQB';

    -- Declare dates from Parameter file
    SELECT
         @StartPerControl = StartPerControl
        ,@EndPerControl   = EndPerControl
        ,@StartDate       = LEFT(StartPerControl,8)
        ,@EndDate         = DATEADD(S,-1,DATEADD(D,1,LEFT(EndPerControl,8)))
        ,@ExportCode      = ExportCode
    FROM dbo.U_dsi_Parameters WITH (NOLOCK)
    WHERE FormatCode = @FormatCode;

    --==========================================
    -- Clean EE List

    --==========================================

    WHERE xCoID <> dbo.dsi_BDM_fn_GetCurrentCOID(xEEID)

    SELECT 
        audEEID  = audKey1Value

        ,audKey3 = audKey3Value
        ,audTableName
        ,audFieldName
        ,audAction
        ,audDateTime

    FROM dbo.vw_AuditData WITH (NOLOCK) 
    JOIN dbo.U_EDRIFORMQB_AuditFields WITH (NOLOCK)

    WHERE audDateTime BETWEEN @StartDate AND @EndDate
    AND audAction <> 'DELETE';

    -- Create Index
    CREATE CLUSTERED INDEX CDX_U_EDRIFORMQB_Audit ON dbo.U_EDRIFORMQB_Audit (audEEID,audKey2);

    --================
    -- Changes Only
    ----================
    --DELETE FROM dbo.U_EDRIFORMQB_EEList
    --WHERE NOT EXISTS (SELECT 1 FROM dbo.U_EDRIFORMQB_Audit WHERE audEEID = xEEID AND audRowNo = 1);

    --==========================================
    -- Create Deduction List
    --==========================================
    DECLARE @DedList VARCHAR(MAX)
    SET @DedList = 'DEN01,DEN02,MEDT,MED02,MED03,VIS01,FSA,MED';

    IF OBJECT_ID('U_EDRIFORMQB_DedList','U') IS NOT NULL
        DROP TABLE dbo.U_EDRIFORMQB_DedList;
    SELECT DISTINCT
         DedCode = DedDedCode
        ,DedType = DedDedType
    INTO dbo.U_EDRIFORMQB_DedList
    FROM dbo.fn_ListToTable(@DedList)
    JOIN dbo.DedCode WITH (NOLOCK)
        ON DedDedCode = Item;

    --==========================================
    -- BDM Section
    --==========================================
    DELETE FROM dbo.U_dsi_BDM_Configuration WHERE FormatCode = @FormatCode;

    -- Required parameters
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'DedCodes','DEN01,DEN02,MEDT,MED02,MED03,VIS01,FSA,MED');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'StartDateTime',@StartDate);
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'EndDateTime',@EndDate);
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES(@FormatCode,'TermSelectionOption','AuditDate');

    -- Non-Required parameters
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'BuildConsolidatedTable','Standard');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'GetChangeReason','Y');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'CobraType','4');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'CobraDate','3');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'UseCobraCoveredDeds','Y');

    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'InvalidCobraReasonsEmp','201,202,204,LEVNT4,205,LEVNT3,208'); -- Invalidate employee when Cobra Reason is a dependent PQB reason â€“ Add valid dependent edhchangereasons (should be same as previous line above)
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'InvalidTermReasonsEmp','203'); -- Invalidate employee when Cobra Reason is "Death".  Add the Death employment term reason (eectermreason)

    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsSpouse','SPS');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsChild','CHL,STC,DPC');
    INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'RelationshipsDomPartner','DP');

    -- Required OE parameters
    IF @ExportCode LIKE '%PASSIVE'
    BEGIN
        INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'OEType','PASSIVE');
    END;

    IF @ExportCode LIKE '%ACTIVE'
    BEGIN
        INSERT INTO dbo.U_dsi_BDM_Configuration VALUES (@FormatCode,'OEType','ACTIVE');
    END;


    -- Run BDM Module
    EXEC dbo.dsi_BDM_sp_PopulateDeductionsTable @FormatCode;

    --==========================================
    -- Build Driver Tables
    --==========================================
    ---------------------------------
    -- DETAIL RECORD - U_EDRIFORMQB_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EDRIFORMQB_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EDRIFORMQB_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvSort = EecEmpNo + CASE WHEN BdmRecType = 'EMP' THEN 'A' ELSE 'B' END
        -- standard fields above and additional driver fields below
        ,drvSSN =  dbo.usi_fn_SSNDash(eepSSN)
        ,drvEmpNo = EecEmpNo
        ,drvDateofBirth = EepDateOfBirth
        ,drvNameLast = EepNameLast
        ,drvNameFirst = EepNameFirst
        ,drvNameMiddle = LEFT(EepNameMiddle,1)
		,drvEmployerName = 'Enverus, Inc.'
        ,drvGender = CASE WHEN EepGender in ('M','F') THEN EepGender  ELSE 'U' END
        ,drvRelation = CASE WHEN conrelationship = 'SPS' THEN 'SPO'
                            WHEN conrelationship = 'DP' THEN 'DOM'
                            WHEN conrelationship in ('CHL','STC','DPC') THEN 'CHL'
                            Else 'EMP' END
        ,drvConSSN = ConSSN
        ,drvConDateofBirth = ConDateOfBirth
        ,drvConNameLast = ConNameLast
        ,drvConNameFirst = ConNameFirst
        ,drvAddress1 = EepAddressLine1
        ,drvAddress2 = EepAddressLine2
        ,drvAddressCity = EepAddressCity
        ,drvAddressState = EepAddressState
        ,drvAddressZipCode = [dbo].[fn_ZipCode](EepAddressZipCode)
        ,drvPhoneHomeNumber = CASE WHEN EepPhoneHomeNumber is null then [dbo].[fn_FormatPhoneNumber](dbo.dsi_fnlib_GetPhoneNumber(EepEEID, 'CEL'),'(###) ###-####') else [dbo].[fn_FormatPhoneNumber](EepPhoneHomeNumber,'(###) ###-####') END
        ,drvBdmBenStopDate = BdmBenStopDate
        ,drvDateofTermination = CASE WHEN EecEmplStatus = 'T' THEN EecDateOfTermination END
        ,drvQualifyingEventCode = CASE    WHEN BdmChangeReason in ('204', 'LEVNT4') THEN 'DIVORCE'
                                        WHEN EecEmplStatus = 'T' and EecTermReason = '203' THEN 'DEATH_OF_EE'  
                                        WHEN BdmChangeReason = '205' THEN 'ENTITLE_TO_MEDICARE' 
                                        WHEN BdmChangeReason in ('201', 'LEVNT3') THEN 'LOSS_OF_ELIGIBILITY_CHILD'
                                        WHEN EecEmplStatus = 'T' and EecTermReason not in ('202','203') and eectermtype = 'V' THEN 'VOLTERM'
                                        WHEN EecEmplStatus = 'T' and EecTermReason not in ('202','203') and eectermtype = 'I' THEN 'INVTERM'
                                        WHEN BdmCobraReason = '208' and EecTermReason = '202' THEN 'RETIREMENT' 
                                        WHEN EecEmplStatus = 'T' and EecTermReason = '202' THEN 'RETIREMENT' 
                                        WHEN BdmChangeReason in ('203','202') THEN 'REDUCTION_OF_HOURS'  
                                        END
		,drvBen1Plan = CASE		WHEN DedCode_MED1 = 'MEDT' THEN 'PPO750'
								WHEN DedCode_MED1 = 'MED02' THEN 'HDHP 2800'
								WHEN DedCode_MED1 = 'MED03' THEN 'HDHP 4000'
								END
        ,drvBen1CoverageCode = CASE WHEN DedCode_MED1 in ('MEDT', 'MED02','MED03') and BenOption_MED1 = 'EE' THEN 'EMP'
                                    WHEN DedCode_MED1 in ('MEDT', 'MED02','MED03') and BenOption_MED1 = 'EES' THEN 'ES'
                                    WHEN DedCode_MED1 in ('MEDT', 'MED02','MED03') and BenOption_MED1 = 'EEC' THEN 'ECH'
                                    WHEN DedCode_MED1 in ('MEDT', 'MED02','MED03') and BenOption_MED1 = 'EEF' THEN 'EFAM'
                                    END
		,drvBen2Plan = CASE		WHEN DedCode_DEN = 'DEN01' THEN 'Base'
								WHEN DedCode_DEN = 'DEN02' THEN 'Buy-Up'
								END
        ,drvBen2CoverageCode =     CASE WHEN DedCode_DEN in ('DEN01','DEN02') and BenOption_DEN = 'EE' THEN 'EMP'
                                    WHEN DedCode_DEN in ('DEN01','DEN02') and BenOption_DEN = 'EES' THEN 'ES'
                                    WHEN DedCode_DEN in ('DEN01','DEN02') and BenOption_DEN = 'EEC' THEN 'ECH'
                                    WHEN DedCode_DEN in ('DEN01','DEN02') and BenOption_DEN = 'EEF' THEN 'EFAM'
                                    END
        ,drvBen3CoverageCode = CASE WHEN DedCode_VIS  = 'VIS01' and BenOption_VIS = 'EE' THEN 'EMP'
                                    WHEN DedCode_VIS  = 'VIS01' and BenOption_VIS = 'EES' THEN 'ES'
                                    WHEN DedCode_VIS  = 'VIS01' and BenOption_VIS = 'EEC' THEN 'ECH'
                                    WHEN DedCode_VIS  = 'VIS01' and BenOption_VIS = 'EEF' THEN 'EFAM'
                                    END
        ,drvFSAIndicator = CASE WHEN DedCode_FSA  = 'FSA' THEN 'X' END
        ,drvFSAMonthlyAmt = CASE When PgrPayFrequency = 'USDIBW' THEN FORMAT(BdmEEAmt*2, '###.##') END
        ,drvMedicalPlanStartDate = CASE WHEN DedCode_MED1 in ('MEDT', 'MED02','MED03') THEN BenStart_MED1 END
    INTO dbo.U_EDRIFORMQB_drvTbl
    FROM dbo.U_EDRIFORMQB_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN (
            SELECT BdmRecType, BdmEEID, BdmCOID, BdmDepRecId, BdmRunId
                ,MAX(BdmBenStartDate) AS BdmBenStartDate
                ,MAX(BdmBenStopDate) AS BdmBenStopDate
                ,MAX(BdmCobraReason) AS BdmCobraReason
                ,MAX(BdmChangeReason) AS BdmChangeReason
                ,MAX(BdmIsPQB) AS BdmIsPQB
                ,MAX(BdmEEAmt) AS BdmEEAmt
                ,MAX(CASE WHEN BdmDedCode in ('VIS01') THEN BdmDedCode END) AS DedCode_VIS 
				,MAX(CASE WHEN BdmDedCode in ('VIS01') THEN BdmBenOption END) AS BenOption_VIS 
                ,MAX(CASE WHEN BdmDedCode in ('DEN01','DEN02')THEN BdmDedCode END) AS DedCode_DEN
                ,MAX(CASE WHEN BdmDedCode in ('DEN01','DEN02') THEN BdmBenOption END) AS BenOption_DEN
                ,MAX(CASE WHEN BdmDedCode in ('FSA') THEN BdmDedCode END) AS DedCode_FSA   
				,MAX(CASE WHEN BdmDedCode in ('FSA') THEN BdmBenOption END) AS BenOption_FSA 
                ,MAX(CASE WHEN BdmDedCode in ('MEDT', 'MED02','MED03') THEN BdmDedCode END) AS DedCode_MED1
                ,MAX(CASE WHEN BdmDedCode in ('MEDT', 'MED02','MED03') THEN BdmBenOption END) AS BenOption_MED1
                ,MAX(CASE WHEN BdmDedCode in ('MEDT', 'MED02','MED03') THEN BdmBenStartDate END) AS BenStart_MED1 
            FROM dbo.U_dsi_BDM_EDRIFORMQB WITH (NOLOCK)
            GROUP BY BdmRecType, BdmEEID, BdmCOID, BdmDepRecId, BdmRunId
        ) AS BDM
        ON BdmEEID = xEEID 
        AND BdmCoID = xCoID
    LEFT JOIN dbo.Contacts WITH (NOLOCK)
        ON ConEEID = xEEID
        AND ConSystemID = BdmDepRecID  
    LEFT JOIN dbo.PayGroup WITH (NOLOCK)
        ON pgrPayGroup = EecPayGroup
    --LEFT JOIN (
 --           SELECT EdhEEID, EdhCOID, EdhEffDate,edhChangeReason, BchIsCobraQualifiedEvent 
 --           FROM (
 --                   SELECT EdhEEID, EdhCOID, EdhEffDate,edhChangeReason, BchIsCobraQualifiedEvent, ROW_NUMBER() OVER (PARTITION by EdhEEID, EdhCOID, edhChangeReason ORDER BY EdhEffDate desc) AS RN
 --                   FROM dbo.EmpHDed a WITH (NOLOCK)
 --                   JOIN dbo.BenChRsn WITH (NOLOCK)
 --                       ON EdhChangeReason = BchCode
 --                   WHERE EdhDedCode IN ('1000M', '1001M', '1005M', '1200D', '1201V')
 --                       AND edhChangeReason IN  ('200','201','202','203','204','205','208', 'LEVNT3','LEVNT4')  AND EdhEffDate BETWEEN @StartDate AND @EndDate 
 --               ) AS X
 --           WHERE RN = 1) AS Edh
 --       ON EdhEEID = xEEID
 --       AND EdhCoID = xCOID  
 --   WHERE BdmRunID = 'QB' 
    ;

    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable(@FormatCode,'UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable(@FormatCode,'Testing') = 'Y' THEN 'Test_Filename_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                                  ELSE 'Filename_' + CONVERT(VARCHAR(8),GETDATE(),112) + '.txt'
                             END
        WHERE FormatCode = @FormatCode;
    END

END;
