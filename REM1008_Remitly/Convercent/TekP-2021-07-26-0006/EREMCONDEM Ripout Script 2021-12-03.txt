        DROP TABLE dbo.U_EREMCONDEM_AuditFields;
    CREATE TABLE dbo.U_EREMCONDEM_AuditFields (aTableName varchar(30),aFieldName varchar(30));

    INSERT INTO dbo.U_EREMCONDEM_AuditFields VALUES ('EmpComp','EecDateofTermination');
    INSERT INTO dbo.U_EREMCONDEM_AuditFields VALUES ('EmpComp','EecEmplStatus');
    INSERT INTO dbo.U_EREMCONDEM_AuditFields VALUES ('EmpComp','EecTermReason');

    -- Create audit table based on fields defined above
    IF OBJECT_ID('U_EREMCONDEM_Audit','U') IS NOT NULL
        DROP TABLE dbo.U_EREMCONDEM_Audit;
    SELECT 
        audEEID  = audKey1Value
        ,audKey2 = audKey2Value
        ,audKey3 = audKey3Value
        ,audTableName
        ,audFieldName
        ,audAction
        ,audDateTime
        ,audOldValue
        ,audNewValue
        ,audRowNo = ROW_NUMBER() OVER (PARTITION BY audKey1Value, audKey2Value, audKey3Value, audFieldName ORDER BY audDateTime DESC)
    INTO dbo.U_EREMCONDEM_Audit
    FROM dbo.vw_AuditData WITH (NOLOCK) 
    JOIN dbo.U_EREMCONDEM_AuditFields WITH (NOLOCK) 
        ON audTableName = aTableName
        AND audFieldName = aFieldName
    WHERE audDateTime BETWEEN dateadd(D, -30, @EndDate) AND @EndDate
    AND audAction <> 'DELETE';

    -- Create Index
    CREATE CLUSTERED INDEX CDX_U_EREMCONDEM_Audit ON dbo.U_EREMCONDEM_Audit (audEEID,audKey2);

    --================
    -- Changes Only
    --================
    IF @Exportcode IN ('CONVDEMCHG')
    BEGIN
        -- Remove Employees with No Changes in Audit
    DELETE FROM dbo.U_EREMCONDEM_EEList
    WHERE NOT EXISTS (SELECT 1 FROM dbo.U_EREMCONDEM_Audit WHERE audEEID = xEEID AND audRowNo = 1);
    END;

    --==========================================
    -- Build Driver Tables
    --==========================================
    ---------------------------------
    -- DETAIL RECORD - U_EREMCONDEM_drvTbl
    ---------------------------------
    IF OBJECT_ID('U_EREMCONDEM_drvTbl','U') IS NOT NULL
        DROP TABLE dbo.U_EREMCONDEM_drvTbl;
    SELECT DISTINCT
         drvEEID = xEEID
        ,drvCoID = xCoID
        ,drvDepRecID = CONVERT(varchar(12),'1') --DELETE IF NOT USING DEPENDENT DATA
        ,drvEmpNoSort = EecEmpNo
        -- standard fields above and additional driver fields below
        ,drvEmploymentIdentifier = substring(eecEmpNo, patindex('%[^0]%',eecEmpNo), len(eecempno)) 
        ,drvUserName = EepAddressEMail
        ,drvIsActive = CASE WHEN EecEmplStatus = 'T' THEN 'false' else 'true' END
        ,drvEmploymentType = CASE WHEN EecEEType = 'REG' and EecFullTimeOrPartTime = 'F' THEN 'Fulltime'
                                  WHEN EecEEType = 'REG' and EecFullTimeOrPartTime = 'P' THEN 'Parttime'
                                  WHEN EecEEType = 'INT' THEN 'Intern'
                                  WHEN EecEEType = 'TMP' THEN 'Temporary'
                                  WHEN EecEEType IN ('CON', 'CST') THEN 'Contract'
                                  WHEN EecEEType = 'Z' THEN 'None'
                            END
        ,drvNameSuffix = NULLIF(EepNameSuffix,'Z')
        ,drvFirstName = CASE WHEN ISNULL(EepNamePreferred,'') <> '' THEN EepNamePreferred ELSE EepNameFirst END
        ,drvMiddleName = LEFT(EepNameMiddle,1)
        ,drvLastName = EepNameLast
        ,drvBusinessTitle = QuoteName(RTRIM(JbcLongDesc),'"')
        ,drvJobCode = EecJobCode
        ,drvJobTitle = QuoteName(JbcDesc,'"')
        ,drvHireDate = EecDateOfLastHire
        ,drvTerminationDate = CASE WHEN EecEmplStatus = 'T' THEN EecDateOfTermination END
        ,drvContactEmail = EepAddressEMail
        ,drvDepartmentIdentifier = EecOrgLvl1
        ,drvDepartmentName = Quotename((Select Top 1 OrgDesc from dbo.OrgLevel WITH (NOLOCK) where OrgCode =  EecOrgLvl1 and OrgLvl = 1),'"')
        ,drvLocationIdentifier = EecLocation
        ,drvLocationName = QuoteName((Select Top 1 LocDesc from dbo.Location WITH (NOLOCK)  where LocCode = EecLocation),'"')
        ,drvLegalEntityIdentifier = EecOrgLvl2
        ,drvLegalEntityName = QuoteName((Select Top 1 OrgDesc from dbo.OrgLevel WITH (NOLOCK) where OrgCode =  EecOrgLvl2 and OrgLvl = 2),'"')
        ,drvSeniorityDate = EecDateOfSeniority
        ,drvSupervisorIdentifier = substring(dbo.dsi_fnlib_GetSupervisorField('', EecEEID, 'EmpNo'), patindex('%[^0]%',dbo.dsi_fnlib_GetSupervisorField('', EecEEID, 'EmpNo')), len(dbo.dsi_fnlib_GetSupervisorField('', EecEEID, 'EmpNo')))
        ,drvBirthdate = EepDateOfBirth
        ,drvGender = CASE WHEN EepGender in ('M') THEN 'Male'
                          WHEN EepGender in ('F') THEN 'Female'
                          ELSE 'Unspecified'
                     END
    INTO dbo.U_EREMCONDEM_drvTbl
    FROM dbo.U_EREMCONDEM_EEList WITH (NOLOCK)
    JOIN dbo.vw_int_EmpComp WITH (NOLOCK)
        ON EecEEID = xEEID 
        AND EecCoID = xCoID
        AND eecEEType <> 'TES'
        and (eecemplstatus <> 'T' OR (eecemplstatus= 'T' and eectermreason <>'TRO' 
                 and EXISTS(select 1 from dbo.U_EREMCONDEM_Audit where audEEID = xEEID AND audKey2 = xcoid and audfieldname = 'eecemplstatus' and audNewValue = 'T')))
    JOIN dbo.EmpPers WITH (NOLOCK)
        ON EepEEID = xEEID
    JOIN dbo.JobCode WITH (NOLOCK)
        ON JbcJobCode = EecJobCode
    ;

    --==========================================
    -- Set FileName
    --==========================================
    IF (dbo.dsi_fnVariable(@FormatCode,'UseFileName') = 'N')
    BEGIN
        UPDATE dbo.U_dsi_Parameters
            SET ExportFile = CASE WHEN dbo.dsi_fnVariable(@FormatCode,'Testing') = 'Y' THEN 'Convercent_Test.csv'
                                  ELSE 'Convercent.csv'
                             END
        WHERE FormatCode = @FormatCode;
    END

END;
